    <script>
        let studentsLoaded = false;
        let facultyLoaded = false;
        let coursesLoaded = false;

        function showTab(tabName) {
            document.querySelectorAll('.tab-content').forEach(tab => {
                tab.classList.remove('active');
            });
            document.querySelectorAll('.tab-btn').forEach(btn => {
                btn.classList.remove('active');
            });

            document.getElementById(tabName).classList.add('active');
            event.target.classList.add('active');

            if (tabName === 'students' && !studentsLoaded) {
                loadStudents();
            } else if (tabName === 'faculty' && !facultyLoaded) {
                loadFaculty();
            } else if (tabName === 'courses' && !coursesLoaded) {
                loadCourses();
            }
        }

        async function loadStats() {
            try {
                const response = await fetch('/api/admin/stats');
                const result = await response.json();
                
                if (result.success) {
                    document.getElementById('totalStudents').textContent = result.data.students;
                    document.getElementById('totalFaculty').textContent = result.data.faculty;
                    document.getElementById('totalCourses').textContent = result.data.courses;
                    document.getElementById('totalEnrollments').textContent = result.data.enrollments;
                }
            } catch (error) {
                console.error('Error loading stats:', error);
            }
        }

        async function loadStudents() {
            try {
                const response = await fetch('/api/admin/students');
                const result = await response.json();
                
                if (result.success) {
                    const students = result.data;
                    if (students.length === 0) {
                        document.getElementById('studentsData').innerHTML = '<p>No students found.</p>';
                    } else {
                        let html = `
                            <table>
                                <thead>
                                    <tr>
                                        <th>ID</th>
                                        <th>Name</th>
                                        <th>Email</th>
                                        <th>Class</th>
                                        <th>Enrollment Date</th>
                                    </tr>
                                </thead>
                                <tbody>
                        `;
                        
                        students.forEach(student => {
                            html += `
                                <tr>
                                    <td>${student.id}</td>
                                    <td>${student.firstName} ${student.lastName}</td>
                                    <td>${student.email}</td>
                                    <td>${student.class || 'N/A'}</td>
                                    <td>${student.enrollmentDate || 'N/A'}</td>
                                </tr>
                            `;
                        });
                        
                        html += '</tbody></table>';
                        document.getElementById('studentsData').innerHTML = html;
                    }
                    studentsLoaded = true;
                }
            } catch (error) {
                document.getElementById('studentsData').innerHTML = '<p>Error loading students</p>';
            }
        }

        async function loadFaculty() {
            try {
                const response = await fetch('/api/admin/faculty');
                const result = await response.json();
                
                if (result.success) {
                    const faculty = result.data;
                    
                    // Populate faculty select for courses
                    let selectHtml = '<option value="">-- Not Assigned --</option>';
                    faculty.forEach(f => {
                        selectHtml += `<option value="${f.id}">Prof. ${f.firstName} ${f.lastName}</option>`;
                    });
                    document.getElementById('facultySelect').innerHTML = selectHtml;
                    
                    if (faculty.length === 0) {
                        document.getElementById('facultyData').innerHTML = '<p>No faculty found.</p>';
                    } else {
                        let html = `
                            <table>
                                <thead>
                                    <tr>
                                        <th>ID</th>
                                        <th>Name</th>
                                        <th>Email</th>
                                        <th>Department</th>
                                        <th>Joining Date</th>
                                    </tr>
                                </thead>
                                <tbody>
                        `;
                        
                        faculty.forEach(f => {
                            html += `
                                <tr>
                                    <td>${f.id}</td>
                                    <td>Prof. ${f.firstName} ${f.lastName}</td>
                                    <td>${f.email}</td>
                                    <td>${f.department || 'N/A'}</td>
                                    <td>${f.joiningDate || 'N/A'}</td>
                                </tr>
                            `;
                        });
                        
                        html += '</tbody></table>';
                        document.getElementById('facultyData').innerHTML = html;
                    }
                    facultyLoaded = true;
                }
            } catch (error) {
                document.getElementById('facultyData').innerHTML = '<p>Error loading faculty</p>';
            }
        }

        async function loadCourses() {
            try {
                const response = await fetch('/api/admin/courses');
                const result = await response.json();
                
                if (result.success) {
                    const courses = result.data;
                    if (courses.length === 0) {
                        document.getElementById('coursesData').innerHTML = '<p>No courses found.</p>';
                    } else {
                        let html = `
                            <table>
                                <thead>
                                    <tr>
                                        <th>ID</th>
                                        <th>Course Name</th>
                                        <th>Code</th>
                                        <th>Credits</th>
                                        <th>Faculty</th>
                                    </tr>
                                </thead>
                                <tbody>
                        `;
                        
                        courses.forEach(course => {
                            html += `
                                <tr>
                                    <td>${course.id}</td>
                                    <td>${course.name}</td>
                                    <td>${course.code}</td>
                                    <td>${course.credits}</td>
                                    <td>${course.faculty}</td>
                                </tr>
                            `;
                        });
                        
                        html += '</tbody></table>';
                        document.getElementById('coursesData').innerHTML = html;
                    }
                    coursesLoaded = true;
                }
            } catch (error) {
                document.getElementById('coursesData').innerHTML = '<p>Error loading courses</p>';
            }
        }

        function openAddStudentModal() {
            document.getElementById('addStudentModal').classList.add('active');
        }

        function openAddFacultyModal() {
            if (!facultyLoaded) loadFaculty();
            document.getElementById('addFacultyModal').classList.add('active');
        }

        function openAddCourseModal() {
            if (!facultyLoaded) loadFaculty();
            document.getElementById('addCourseModal').classList.add('active');
        }

        function closeModal(modalId) {
            document.getElementById(modalId).classList.remove('active');
        }

        // Add Student Form
        document.getElementById('addStudentForm').addEventListener('submit', async function(e) {
            e.preventDefault();
            const formData = new FormData(e.target);
            const data = Object.fromEntries(formData);

            try {
                const response = await fetch('/api/admin/students', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(data)
                });

                const result = await response.json();
                const alertDiv = document.getElementById('studentAlert');

                if (result.success) {
                    alertDiv.innerHTML = '<div class="alert alert-success">Student added successfully!</div>';
                    closeModal('addStudentModal');
                    e.target.reset();
                    loadStudents();
                    loadStats();
                } else {
                    alertDiv.innerHTML = '<div class="alert alert-error">Error: ' + result.message + '</div>';
                }
            } catch (error) {
                document.getElementById('studentAlert').innerHTML = '<div class="alert alert-error">Error adding student</div>';
            }
        });

        // Add Faculty Form
        document.getElementById('addFacultyForm').addEventListener('submit', async function(e) {
            e.preventDefault();
            const formData = new FormData(e.target);
            const data = Object.fromEntries(formData);

            try {
                const response = await fetch('/api/admin/faculty', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(data)
                });

                const result = await response.json();
                const alertDiv = document.getElementById('facultyAlert');

                if (result.success) {
                    alertDiv.innerHTML = '<div class="alert alert-success">Faculty added successfully!</div>';
                    closeModal('addFacultyModal');
                    e.target.reset();
                    loadFaculty();
                    loadStats();
                } else {
                    alertDiv.innerHTML = '<div class="alert alert-error">Error: ' + result.message + '</div>';
                }
            } catch (error) {
                document.getElementById('facultyAlert').innerHTML = '<div class="alert alert-error">Error adding faculty</div>';
            }
        });

        // Add Course Form
        document.getElementById('addCourseForm').addEventListener('submit', async function(e) {
            e.preventDefault();
            const formData = new FormData(e.target);
            const data = Object.fromEntries(formData);
            if (!data.facultyId) delete data.facultyId;

            try {
                const response = await fetch('/api/admin/courses', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(data)
                });

                const result = await response.json();
                const alertDiv = document.getElementById('courseAlert');

                if (result.success) {
                    alertDiv.innerHTML = '<div class="alert alert-success">Course added successfully!</div>';
                    closeModal('addCourseModal');
                    e.target.reset();
                    loadCourses();
                    loadStats();
                } else {
                    alertDiv.innerHTML = '<div class="alert alert-error">Error: ' + result.message + '</div>';
                }
            } catch (error) {
                document.getElementById('courseAlert').innerHTML = '<div class="alert alert-error">Error adding course</div>';
            }
        });

        // Load initial data
        loadStats();
        loadStudents();
    </script>
